//countDownBegin
let rec countDown (n: Nat): Nat list =
    if n = 0N then [0N]
    else n :: countDown (n - 1N)
//countDownEnd


//countUpBegin
let rec countUp (n: Nat): Nat list =
    if n = 0N then [0N]
    else countUp (n - 1N) @ [n]
//countUpEnd


//factorialBegin
let rec factorial (n: Nat): Nat =
    if n = 0N then 1N
    else n * factorial (n - 1N)
//factorialEnd


//mod5Begin
let rec mod5 (n: Nat): Nat =
    if n = 0N then 0N
    else
        let m: Nat = mod5 (n - 1N)
        if m = 4N then 0N
        else m + 1N
//mod5End


//sumUpBegin
let rec sumUp (n: Nat): Nat =
    if n = 0N then 0N
    else n + sumUp (n - 1N)
//sumUpEnd


//powerBegin
let rec power (baseValue: Nat) (n: Nat): Nat =
    if n = 0N then 1N
    else baseValue * power baseValue (n - 1N)
//powerEnd


//isEvenBegin
let rec isEven (n: Nat): bool =
    if n = 0N then true
    else not (isEven (n - 1N))
//isEvenEnd
