\makeatletter
\newcommand\sectionmarker{\faIcon{tasks}}
\renewcommand\thesection{Aufgabe\enskip\arabic{section}:\enskip}
\renewcommand*{\sectionformat}{\sectionmarker\enskip\thesection\enskip}
\renewcommand{\sectionmark}[1]{\gdef\rightmark{#1}}
\let\oldsection\section
\RenewDocumentCommand{\section}{s o m}{%
	\IfBooleanTF{#1}%
		{%
			\IfValueTF{#2}%
				{\oldsection*{\sectionmarker~#3}\markboth{#2}{}}%
				{\oldsection*{\sectionmarker~#3}}%
		}%
		{\IfValueTF{#2}{\oldsection[#2]{#3}}{\oldsection{#3}}}%
}
\makeatother

}
