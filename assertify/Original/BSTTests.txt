type ValidBST<'a> = Valid of BST<'a> * List<'a> * Nat // (BST, BST als Liste, Höhe des BSTs)
type InvalidBST<'a> = Invalid of BST<'a>

// need this for type inference purposes in `natRange`
let makeNat (n : int) : Nat = Nat.Make n
// need this for type inference purposes in `AssertEquals`
let empty<'a, 'b> : BSTreeish<'a,'b> = Emptyish

[<StructuralEquality;StructuralComparison>]
type Extended<'a when 'a: comparison> =
    | NegInfty
    | Just of 'a
    | PosInfty

let natRange (lo : Nat) (hi : Nat) : Gen<Nat> = Gen.map makeNat (Gen.choose (int lo , int hi))

let genRange : (Extended<Nat> * Extended<Nat>) -> Gen<Nat> = function
    | Just lo , Just hi -> natRange lo hi
    | NegInfty, Just hi -> natRange 0N hi
    | NegInfty, PosInfty-> Arb.generate
    | Just lo, PosInfty -> Gen.map (fun x -> x + lo) Arb.generate
    | _ -> failwith "purposely no pattern for(_,-∞)"


let isBST<'a when 'a: comparison> (root: BST<'a>): Bool =
 let rec bounds (root: BST<'a>) : Option<Range<'a>> =
  match root with
  | Empty -> Some(EmptyR)
  | Node(left, p, right) ->
   match (bounds left, bounds right) with
   | Some EmptyR      , Some EmptyR       -> Some(Twixt(p, p))
   | Some EmptyR      , Some(Twixt(c, d)) -> if p <= c then Some(Twixt(p, d)) else None
   | Some(Twixt(a, b)), Some EmptyR       -> if b <= p then Some(Twixt(a, p)) else None
   | Some(Twixt(a, b)), Some(Twixt(c, d)) ->
       if b <= p && p <= c then Some(Twixt(a, d)) else None
   | _                                    -> None
 in match bounds root with
    | None -> false
    | _    -> true

let rec deleteMin<'a> (root: BST<'a>): Option<'a * BST<'a>> =
    match root with
    | Empty -> None
    | Node(l, x, r) ->
        match deleteMin l with
        | None -> Some (x, r) // l = Empty
        | Some(a, lMinusA) -> Some(a, Node(lMinusA, x, r))

let rec tracer (grow : 'b -> Option<'a * 'b>) (seed : 'b) : List<'a> =
    match grow seed with
    | None -> []
    | Some(a, seed') -> a :: tracer grow seed'

let rec toList<'a when 'a: comparison> (root: BST<'a>): List<'a> =
    tracer deleteMin root

let rec genAnyTree size =
    if size <= 0 then Gen.constant Empty
    else
        gen {
            let! root = Arb.generate
            let! sizeL = Gen.choose(0, size - 1)
            let! sizeR = Gen.choose(0, size - 1)
            let! treeL = genAnyTree sizeL
            let! treeR = genAnyTree sizeR
            return Node (treeL, root, treeR)
        }

// Generator für gültigen BST mit gegebener Tiefe und Range die die Werte annehmen können
let rec generateValidBST (size : int) (range : Extended<Nat> * Extended<Nat>) =
    if size <= 0 then Gen.constant (Empty, [], 0N)
    else (gen {
     let (lo, hi) = range
     let! p = genRange(range)
     let! sizeL = Gen.choose(0, size - 1)
     let! sizeR = Gen.choose(0, size - 1)
     let! (treeL, listL, heightL) = generateValidBST sizeL (lo, (Just p))
     let! (treeR, listR, heightR) = generateValidBST sizeR ((Just p), hi)
     return (
                Node (treeL, p, treeR),
                listL @ [p] @ listR,
                1N + max heightL heightR
            )     
        
    })

type ArbitraryModifiers =
    static member Nat() =
        Arb.from<NonNegativeInt>        
        |> Arb.convert (fun i -> Nat.Make (int i)) (fun n -> NonNegativeInt (int n))

    static member ValidBST() =
        Arb.fromGen << Gen.sized <| fun size ->
            gen {
                let! result = generateValidBST size (NegInfty, PosInfty)
                return Valid result
            }

    static member InvalidBST() =
        Arb.fromGen << Gen.map Invalid << Gen.filter (fun p -> not (isBST p)) << Gen.sized <| genAnyTree

let mutable counter: int = 0;

[<CustomEquality>] [<CustomComparison>]
type ComparisonCount<'a when 'a :> IComparable<'a> and 'a: equality> =
    CC of 'a
    with
    static member wrap (value: 'a): ComparisonCount<'a> = CC value
    static member unwrap (x: ComparisonCount<'a>): 'a = let (CC me) = x in me
    interface IComparable<ComparisonCount<'a>> with
        member this.CompareTo (CC other) =
            counter <- counter + 1
            (ComparisonCount.unwrap this :> IComparable<'a>).CompareTo other
    interface IComparable with
        member this.CompareTo obj =
            match obj with
            | null -> 1
            | :? ComparisonCount<'a> as other -> (this :> IComparable<_>).CompareTo other
            | _ -> invalidArg "obj" "not a ComparisonCount<'a>"
    override this.Equals obj =
        match obj with
        | :? ComparisonCount<'a> as other -> ComparisonCount.unwrap this = ComparisonCount.unwrap other
        | _ -> false
    override this.GetHashCode() = hash (ComparisonCount.unwrap this)


[<TestClass>]
type Tests() =
    do Arb.register<ArbitraryModifiers>() |> ignore

    let ex1 = Node(Node(Empty,1N,Empty), 2N, Node(Empty,4N,Empty))
    let ex2 = Node(Node(Empty,2N,Empty), 3N, Node(Empty,5N,Empty))
    let ex3 = Node(Node(Empty,1N,Empty), 1N, Empty)
    let inv1 = Node(Node(Empty,3N,Empty), 2N, Empty)
    let inv2 = Node(Node(Node(Empty,4N,Empty), 2N, Empty), 3N, Empty)

    // ------------------------------------------------------------------------
    // a)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``size Beispiel 1`` (): unit =
        Assert.AreEqual<Nat>(0N, BSTs.size Empty)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``size Beispiel 2`` (): unit =
        Assert.AreEqual<Nat>(3N, BSTs.size ex1)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``size Beispiel 3`` (): unit =
        Assert.AreEqual<Nat>(2N, BSTs.size ex3)

    [<TestMethod>] [<Timeout(10000)>]
    member _.``size Zufall`` (): unit =
        Check.One({Config.QuickThrowOnFailure with MaxTest = 100; EndSize = 5}, fun (Valid (bst, list, _): ValidBST<Nat>) ->
            Assert.AreEqual<Nat>(Nat.Make(list.Length), BSTs.size bst)
        )

    [<TestMethod>] [<Timeout(1000)>]
    member _.``height Beispiel 1`` (): unit =
        Assert.AreEqual<Nat>(0N, BSTs.height Empty)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``height Beispiel 2`` (): unit =
        Assert.AreEqual<Nat>(2N, BSTs.height ex1)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``height Beispiel 3`` (): unit =
        Assert.AreEqual<Nat>(2N, BSTs.height ex3)

    [<TestMethod>] [<Timeout(10000)>]
    member _.``height Zufall`` (): unit =
        Check.One({Config.QuickThrowOnFailure with MaxTest = 100; EndSize = 5}, fun (Valid (bst, _, h): ValidBST<Nat>) ->
            Assert.AreEqual<Nat>(h, BSTs.height bst)
        )


    // ------------------------------------------------------------------------
    // b)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``isBST Beispiel 1`` (): unit =
        Assert.IsTrue(BSTs.isBST<Nat> Empty)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``isBST Beispiel 2`` (): unit =
        Assert.IsTrue(BSTs.isBST ex1)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``isBST Beispiel 3`` (): unit =
        Assert.IsTrue(BSTs.isBST ex2)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``isBST Beispiel 4`` (): unit =
        Assert.IsTrue(BSTs.isBST ex3)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``isBST Beispiel 5`` (): unit =
        Assert.IsFalse(BSTs.isBST inv1)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``isBST Beispiel 6`` (): unit =
        Assert.IsFalse(BSTs.isBST inv2)

    [<TestMethod>] [<Timeout(60000)>]
    member _.``isBST Zufall Gültig`` (): unit =
        Check.One({Config.QuickThrowOnFailure with MaxTest = 1000; EndSize = 5}, fun (Valid (bst, _, _): ValidBST<Nat>) ->
            Assert.IsTrue(
                BSTs.isBST bst,
                "BST wurde nicht als gültig erkannt."
            )
        )

    [<TestMethod>] [<Timeout(60000)>]
    member _.``isBST Zufall Ungültig`` (): unit =
        Check.One({Config.QuickThrowOnFailure with MaxTest = 1000; EndSize = 5}, fun (Invalid bst: InvalidBST<Nat>) ->
            Assert.IsFalse(
                BSTs.isBST bst,
                "BST wurde nicht als ungültig erkannt."
            )
        )


    // ------------------------------------------------------------------------
    // c)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``deleteMin leer`` (): unit =
        Assert.AreEqual(None, BSTs.deleteMin Empty)

    [<TestMethod>] [<Timeout(20000)>]
    member _.``deleteMin Zufall`` (): unit =
        Check.One({Config.QuickThrowOnFailure with StartSize = 1; EndSize = 5; MaxTest = 30}, fun (Valid (bst, _, _): ValidBST<Nat>) ->
            match BSTs.deleteMin bst with
            | None -> failwith "Aus nicht-leerem Baum konnte nicht gelöscht werden"
            | Some(n, rest) ->
                  match deleteMin bst with | Some(m, restm) ->
                  Assert.AreEqual<Nat>(m, n)
                  Assert.AreEqual(toList restm, toList rest)
        )

    // ------------------------------------------------------------------------
    // d)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``partition leer`` (): unit =
        Assert.AreEqual<BSTreeish<Nat,List<Nat>>>(empty<Nat,List<Nat>>, BSTs.partition<Nat> [])

    [<TestMethod>][<Timeout(20000)>]
    member this.``partition Zufall``() : unit =
        Check.One(
            { Config.QuickThrowOnFailure with
                StartSize = 1
                EndSize = 5
                MaxTest = 30 },
            fun (NonEmptyArray ar) ->
            let (list : List<Nat>) = List.ofArray ar
            match BSTs.partition list with
            | Emptyish -> failwith "Nicht-leere Liste konnte nicht zweigeteilt werden"
            | Nodeish(l, p, r) ->
                Assert.AreEqual<Nat>(List.last list, p)
                Assert.AreEqual(List.sort list, List.sort (l @ [ p ] @ r), "Elemente nicht mehr dieselben")
                Assert.IsTrue(List.forall (fun x -> x <= p) l, "Elemente links nicht alle <=")
                Assert.IsTrue(List.forall (fun x -> x >= p) r, "Elemente rechts nicht alle >=")
        )


    // ------------------------------------------------------------------------
    // e)

    [<TestMethod>] [<Timeout(10000)>]
    member _.``letIt Beispiele`` (): unit =
        let fibTree = fun n -> if n = 0N then Emptyish else Nodeish(n - 1N, n, n - 2N) in
        test <@ BSTs.letIt fibTree 3N = Node(Node (Node (Empty, 1N, Empty), 2N, Empty), 3N, Node (Empty, 1N, Empty)) @>
        test <@ BSTs.letIt fibTree 4N = Node (Node        (Node (Node (Empty, 1N, Empty), 2N, Empty), 3N, Node (Empty, 1N, Empty)),      4N, Node (Node (Empty, 1N, Empty), 2N, Empty)) @>

    // ------------------------------------------------------------------------
    // f)

    [<TestMethod>] [<Timeout(20000)>]
    member _.``toList Zufall`` (): unit =
        Check.One({Config.QuickThrowOnFailure with EndSize = 5; MaxTest = 30}, fun (Valid (bst, list, _): ValidBST<Nat>) ->
            Assert.AreEqual(list, BSTs.toList bst)
        )


    // ------------------------------------------------------------------------
    // g)

    [<TestMethod>] [<Timeout(1000)>]
    member _.``quickSort Beispiel 1`` (): unit =
        let expected: Nat list = []
        Assert.AreEqual(expected, BSTs.quickSort<Nat> [])

    [<TestMethod>] [<Timeout(1000)>]
    member _.``quickSort Beispiel 2`` (): unit =
        Assert.AreEqual([1N; 2N; 2N; 3N], BSTs.quickSort [2N; 3N; 1N; 2N])

    [<TestMethod>] [<Timeout(10000)>]
    member _.``quickSort Zufall`` (): unit =
        Check.One({Config.QuickThrowOnFailure with MaxTest = 100; EndSize = 50}, fun (xs: Nat list) ->
            Assert.AreEqual(List.sort xs, BSTs.quickSort xs)
        )
