module ListTests.TestsModule

open Microsoft.VisualStudio.TestTools.UnitTesting
open FsCheck
open Swensen.Unquote
open Mini

type ArbitraryModifiers =
// Restrict to range (0,7) so that lists generated for `runs` have a chance of containing runs
    static member Nat() =
        (Gen.map (fun (n : int) -> (bigint n)) (Gen.choose(0,7)),
         Arb.shrink<bigint>)
         |> Arb.fromGenShrink
         |> Arb.convert (Nat.Make) (fun n -> n.ToBigInteger())

// Ugly duplication, should import reference solution as separate module
let rec intersperse<'a> (sep: 'a) (xs: List<'a>): List<'a> =
    match xs with
    | [] -> []
    | one & [_] -> one
    | x :: xs -> x :: sep :: intersperse sep xs

let rec runs<'a when 'a: comparison> (xs: List<'a>): List<List<'a>> =
    match xs with
    | [] -> []
    | x::xs -> match runs xs with
               | rest & ((current & (y :: _)) :: ys) ->
                   if x <= y then (x :: current) :: ys else [x] :: rest
               | rest -> [x] :: rest
// end ugly duplication

let config = {
    Config.QuickThrowOnFailure with
        EndSize = 1000
        MaxTest = 1000
    }

let ex = [2N; 4N; 3N; 4N; 2N; 1N]

[<TestClass>]
type Tests() =
    do Arb.register<ArbitraryModifiers>() |> ignore

    // ------------------------------------------------------------------------
    // a)

    [<TestMethod>] [<Timeout 1000>]
    member this.``a) minAndMax Beispiele`` (): unit =
        test <@ Lists.minAndMax [] = None @>
        test <@ Lists.minAndMax [1N;2N;3N;4N] = Some(1N, 4N) @>
        test <@ Lists.minAndMax [4N;3N;2N;1N] = Some(1N, 4N) @>
        test <@ Lists.minAndMax [4N;7N;1N;1N] = Some(1N, 7N) @>
        test <@ Lists.minAndMax ex = Some(1N, 4N) @>

    [<TestMethod>] [<Timeout 5000>]
    member this.``a) minAndMax Zufallstest`` (): unit =
        Check.QuickThrowOnFailure(fun (xs: List<Nat>)->
            Assert.AreEqual(
                List.fold (fun acc x ->
                    match acc with
                    | None -> Some(x, x)
                    | Some(low, high) -> Some(min x low, max x high)
                ) None xs,
                Lists.minAndMax xs
            )
        )

    // ------------------------------------------------------------------------
    // b)

    [<TestMethod>] [<Timeout 1000>]
    member this.``b) map Beispiele`` (): unit =
        test <@ Lists.map (fun x -> x + 1N) [] = [] @>
        test <@ Lists.map (fun x -> x + 1N) [1N;2N;3N] = [2N;3N;4N] @>
        test <@ Lists.map (fun x -> x * 2N) [1N;2N;3N] = [2N;4N;6N] @>
        test <@ Lists.map (fun x -> x * x) [4N;7N;1N;1N] = [16N;49N;1N;1N] @>
        test <@ Lists.map (fun x -> x + 1N) ex = [3N;5N;4N;5N;3N;2N] @>

    [<TestMethod>] [<Timeout 5000>]
    member this.``b) map Zufallstest`` (): unit =
        Check.QuickThrowOnFailure(fun (p: Nat -> Bool) (xs: List<Nat>) ->
            Assert.AreEqual(
                List.map p xs,
                Lists.map p xs
            )
        )

    // ------------------------------------------------------------------------
    // c)

    [<TestMethod>] [<Timeout 1000>]
    member this.``c) duplicate Beispiele`` (): unit =
        test <@ Lists.duplicate [] = [] @>
        test <@ Lists.duplicate [1N;2N;3N] = [1N;1N;2N;2N;3N;3N] @>
        test <@ Lists.duplicate [4N;7N;1N;1N] = [4N;4N;7N;7N;1N;1N;1N;1N] @>

    [<TestMethod>] [<Timeout 5000>]
    member this.``c) duplicate Zufallstest`` (): unit =
        Check.QuickThrowOnFailure(fun (xs: List<Nat>) ->
            Assert.AreEqual(
                List.foldBack (fun x acc -> x :: x :: acc) xs [],
                Lists.duplicate xs
            )
        )

    // ------------------------------------------------------------------------
    // d)

    [<TestMethod>] [<Timeout 1000>]
    member _.``d) collect Beispiele`` (): unit =
        test <@ Lists.collect (fun x -> [x + 1N; x + 2N]) [] = [] @>
        test <@ Lists.collect (fun x -> [x + 1N; x + 2N]) [1N; 2N; 3N] = [2N; 3N; 3N; 4N; 4N; 5N] @>
        test <@ Lists.collect (fun x -> [x * 2N; x * 3N]) [4N; 7N; 1N; 1N] = [8N; 12N; 14N; 21N; 2N; 3N; 2N; 3N] @>

    [<TestMethod>] [<Timeout 5000>]
    member _.``d) collect Zufallstest``() : unit =
        Check.QuickThrowOnFailure(fun (xs: List<Nat>) ->
            let fG: Arbitrary<Nat -> List<Nat>> =
                (Arb.fromGen << Gen.elements)
                    [ (fun x -> [ x + 1N; x + 2N ])
                      (fun x -> [ x + 1N; x + 2N ])
                      (fun x -> [ x * 2N; x * 3N ]) ] in
            Prop.forAll fG (fun f -> Assert.AreEqual(List.collect f xs, Lists.collect f xs)))

    // ------------------------------------------------------------------------
    // e)

    [<TestMethod>] [<Timeout 1000>]
    member _.``e) intersperse Beispiele`` (): unit =
        test <@ Lists.intersperse 0N [] = [] @>
        test <@ Lists.intersperse 0N [2N] = [2N] @>
        test <@ Lists.intersperse 0N [1N; 2N; 3N; 4N] = [1N; 0N; 2N; 0N; 3N; 0N; 4N] @>

    [<TestMethod>] [<Timeout 5000>]
    member _.``e) intersperse Zufallstest`` (): unit =
        Check.QuickThrowOnFailure(fun (xs: List<Nat>) ->
                                  fun (x : Nat) ->
                                  Assert.AreEqual(intersperse x xs,  Lists.intersperse x xs))

    // ------------------------------------------------------------------------
    // f)

    [<TestMethod>] [<Timeout 1000>]
    member _.``f) runs Beispiele`` (): unit =
        test <@ Lists.runs [] = []  @>
        test <@ Lists.runs [1N; 2N; 3N] = [[1N; 2N; 3N]] @>
        test <@ Lists.runs [3N; 2N; 1N] = [[3N]; [2N]; [1N]] @>
        test <@ Lists.runs [4N; 7N; 1N; 1N] = [[4N; 7N]; [1N; 1N]] @>

    [<TestMethod>] [<Timeout 5000>]
    member _.``f) runs Zufallstest`` (): unit =
        Check.QuickThrowOnFailure(fun (xs: List<Nat>) ->
                                  Assert.AreEqual(runs xs,  Lists.runs xs))
