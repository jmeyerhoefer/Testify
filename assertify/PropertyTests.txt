type SimpleRegExp<'a> =
    | SEps
    | SLit of 'a
    | SCat of SimpleRegExp<'a> * SimpleRegExp<'a>
    | SEmpty
    | SOr of SimpleRegExp<'a> * SimpleRegExp<'a>
    | SStar of SimpleRegExp<'a>

;
let rec private toRegExp<'a> (r: SimpleRegExp<'a>): RegExp<'a> =
    match r with
    | SEmpty -> Empty
    | SEps -> Eps
    | SLit a -> Lit a
    | SCat (r1, r2) -> Cat (toRegExp r1, toRegExp r2)
    | SOr (r1, r2) -> Or (toRegExp r1, toRegExp r2)
    | SStar r1 -> Star (toRegExp r1)


let rec private isWord<'a> (r: RegExp<'a>): 'a list option =
    match r with
    | Eps -> Some []
    | Lit a -> Some [ a ]
    | Cat (r1, r2) -> 
        match isWord r1, isWord r2 with
        | Some l1, Some l2 -> Some (l1 @ l2)
        | _ -> None
    | _ -> None

static member Check' (prop: 'a -> Expr<bool>, ?config: FsCheck.Config, ?gen: Gen<'a>): unit =
        let config: FsCheck.Config = defaultArg config FsCheck.Config.Default
        let captureRunner: CaptureRunner = CaptureRunner config.Runner
        let configuration = config.WithRunner captureRunner

        let arbOpt: Arbitrary<'a> option =
            match gen with
            | None -> None
            | Some g -> Some (FsCheck.FSharp.Arb.fromGen g)

        let mutable failedExpression: Expr<bool> option = None
        let property: FsCheck.Property =
            let test: 'a -> bool = fun (x: 'a) ->
                let expr = prop x
                try
                    if expr.Eval() then true
                    else
                        failedExpression <- Some expr
                        false
                with _ ->
                    failedExpression <- Some expr
                    false
            match arbOpt with
            | Some arb -> FsCheck.FSharp.Prop.forAll arb test
            | None ->  FsCheck.FSharp.Prop.ofTestable test

        try FsCheck.Check.One (configuration, property) with
        | ex ->
            let result: AssertifyResult =
                match failedExpression with
                | Some expr ->
                    let expected, actual: obj option * obj option =
                        match expr with
                        | DerivedPatterns.SpecificCall <@ (=) @> (_, _, [ left; right ]) ->
                            Some (left.Eval ()), Some (right.Eval ())
                        | _ -> None, None
                    Assertify.makeResult (
                        "Check",
                        expression = expr,
                        expected = expected,
                        actual = actual,
                        ?originalInputs = captureRunner.Original,
                        ?shrunkInputs = captureRunner.LastShrink,
                        errorMessage = ex.Message,
                        stacktrace = ex.StackTrace
                    )
                | None ->
                    Assertify.makeResult (
                        "Check",
                        errorMessage = ex.Message,
                        stacktrace = ex.StackTrace
                    )

            Json.serialize result |> failwithf "%s"

                let expr: Expr<bool> = prop x

                    if expr.Eval () then true
                    else
                        failedExpression <- Some expr

                | ex -> false
