%===============================================================================
% TCBSET +++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
%===============================================================================


\tcbset{
    %=======================================================
	% PANEL SETUP ++++++++++++++++++++++++++++++++++++++++++
	%=======================================================
    panel setup/.code={
		\panelColorExplicitfalse%
		\panelUseMintedfalse%
		\def\panelMintedApp{}%

		\def\panelMintedFirstLine{}%
		\def\panelMintedLastLine{}%
		\def\panelMintedHighlightLines{}%
		\def\panelMintedHighlightColor{metroOrange!25}%

		\pgfkeys{/panel/.cd, listing, #1}%
		\ifpanelColorExplicit\else\let\panelColor\panelColorDefault\fi%
		\refstepcounter{\panelCounter}%

		\edef\panelMintedOptExtra{}%
		\ifstrempty{\panelMintedHighlightLines}{}{%
			\edef\panelMintedOptExtra{%
				highlightlines=\panelMintedHighlightLines,%
				highlightcolor=\panelMintedHighlightColor%
			}%
		}%
	},
    %=======================================================
	% SIMPLE BASE ++++++++++++++++++++++++++++++++++++++++++
	%=======================================================
    panel base simple/.style={
		enhanced,
        notitle,
		colback=black!5!white,
		arc=0mm,
		boxrule=0mm,
		toprule=0mm,
		leftrule=0mm, rightrule=0mm,
		top=1mm, bottom=1mm,
		left=2mm, right=2mm,
		borderline west = { 1.5mm }{ -3mm }{ #1 },
		before={ \par\medskip\parindent=0pt }, after={ \par\smallskip },
	},
    panel inline/.style={
        enhanced,
        notitle,
        colback=black!5!white,
        arc=0mm,
		boxrule=0mm,
		leftrule=0mm, rightrule=0mm,
		top=1mm, bottom=1mm,
		left=2mm, right=2mm,
		borderline west = { 1.5mm }{ -3mm }{ #1 },
		before={ \par\medskip\parindent=0pt }, after={ \par\smallskip },
    },
    %=======================================================
	% PANEL BASE +++++++++++++++++++++++++++++++++++++++++++
	%=======================================================
	panel base/.style={
		enhanced,
		attach boxed title to bottom*,
		fonttitle=\sffamily,
		colframe=\panelColor,
		colback=black!5!white,
		colbacktitle=\panelColor!25!white,
		coltitle=black,
		arc=0mm,
		boxrule=0mm,
		toprule=0mm,
		leftrule=0mm, rightrule=0mm,
		toptitle=0.5mm, bottomtitle=0.5mm,
		top=1mm, bottom=1mm,
		left=2mm, right=2mm,
		borderline west = { 1.5mm }{ -3mm }{ \panelColor },
		before={ \par\medskip\parindent=0pt }, after={ \par\smallskip },
        fontlower=\ttfamily\footnotesize,
        colbacklower=green,
	},
    %=======================================================
	% PANEL MINTED +++++++++++++++++++++++++++++++++++++++++
	%=======================================================
    panel minted/.style={
		listing engine=minted,
		minted style=colorful,
		minted language=\panelMintedLanguage,
		minted options={
			fontsize=\footnotesize,
			linenos=true, numbers=right,
            escapeinside=\#\%,
			xrightmargin=-.5em,
			numbersep=0.1pt,
			breaklines=true,
			breaksymbolleft=\tiny\ensuremath{\textcolor{colorLineBreakSymbol}{\hookrightarrow}},
			breakbytokenanywhere=true,
			stripnl=true,
			tabsize=4, autogobble=true,
		},
		minted options app={ \panelMintedApp },
		minted options app/.expanded={ \panelMintedOptExtra },
	},
    %=======================================================
	% PANEL TABLE ++++++++++++++++++++++++++++++++++++++++++
	%=======================================================
    panel table/.style={
        tabularx*={%
            \begingroup%
            \setlength{\arrayrulewidth}{.4pt}%
            \renewcommand\arraystretch{1.4}%
        }{ @{\hspace{5mm}}#1@{\hspace{5mm}} },
        after upper app={ \endgroup },
	},
}

\newtcblisting{codepanel}[2][]{%
	panel setup={ fs, #1 },
	panel base,
    listing only,
	panel minted,
	title={ \strut\textbf{\panelKindText\ \csname the\panelCounter\endcsname}~#2 },
	% label={ #2 },
}

\newtcblisting{outputpanel}[4][]{%
	panel setup={ fs, #1 },
	panel base,
    listing and comment,
    comment={ #4 },
	panel minted,
	title={ \strut\textbf{\panelKindText\ \csname the\panelCounter\endcsname}~#2 },
	label={ code:#3 },
}


%===========================================================

\newtcbinputlisting{\inputcodepanel}[3][]{%
	panel setup={ fs, #1 },
	panel base,
    listing only,
	panel minted,
	listing file=#3,
	minted options app/.expanded={%
		\ifstrempty{\panelMintedFirstLine}{}{firstline=\panelMintedFirstLine,}%
		\ifstrempty{\panelMintedLastLine}{}{lastline=\panelMintedLastLine,}%
	},
	title={ \strut\textbf{\panelKindText\ \csname the\panelCounter\endcsname}~#2 },
	% label={ #2 },
}
