\newcommand{\todo}[1]%
{
	\vspace{0.5em}\par\noindent
	\begin{tcolorbox}%
	[
		% title
		title = TODO,
		fonttitle = \bfseries,
		coltitle = white,
		colbacktitle = blue,
		center title,
		% frame
		colback = blue!20,
		colframe = blue,
		enhanced = true
	]
		\begin{minipage}[t]{\dimexpr\hsize-2\fboxsep-2\fboxrule-10pt\relax}
			\ifx\empty#1\empty
				Apparently there is something to do.
			\else
				#1
			\fi
		\end{minipage}
	\end{tcolorbox}
	\vspace{0.5em}\par\noindent
}
